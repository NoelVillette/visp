cmake_minimum_required (VERSION 3.5)
project(visp)

find_package(pybind11 REQUIRED)
find_package(VISP REQUIRED)
include_directories(${VISP_INCLUDE_DIRS})
include_directories(include)
message(${VISP_DIR})
string(REPLACE ";"  "\n" str "${VISP_INCLUDE_DIRS}")
message(STATUS ${str})
string(REPLACE ";"  "\n" str "${VISP_LIBRARIES}")
message(STATUS ${str})
message(${VISP_VERSION})
message(${pybind11_VERSION})
file(GLOB generated_cpp src/*.cpp)
file(GLOB static_cpp src/static/*.cpp)
pybind11_add_module(visp THIN_LTO ${generated_cpp} ${static_cpp})
target_link_libraries(visp PRIVATE ${VISP_LIBRARIES})
